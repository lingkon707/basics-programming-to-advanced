```javascript

```

```javascript
// Calculate the sine of an angle in degrees (e.g., sin(30))
let angleInDegrees = 30;
let angleInRadians = angleInDegrees * (Math.PI / 180);

console.log("The sine of 30 degrees is:", sineValue);
```
